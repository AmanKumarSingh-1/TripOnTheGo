            echo "You did okay!";

            break;
        case 'F':
            echo "You failed!";
            break;
        default:
            echo "{$grade} is an Invalid grade!";
    }
    echo "<br><br>";

    //          targeted advertisements, browsing preferences, and

    setcookie("fav_food", "pizza", time() + 86400, "/");
    setcookie("fav_drink", "coffee", time() + 86400 * 2, "/");
    setcookie("fav_desert", "icecream", time() + 86400 * 3, "/");

    foreach ($_COOKIE as $key => $value) {
        echo "{$key}: {$value} <br>";
